##### Important commands for lephare #####
#Naming convention: INPUTCAT_SEDlib_FILTERlib_note.out[.prob]
#Replace user_path by your path to lephare folder
#change keywords as you wish
#STAR_PDF_OUT: output file path to store the star PDF in


##### mag_gal (make magnitudes library) #####

###buzzard in LSST filters (for DC1, only gal)
mag_gal  -c lephare/training_stats/simulation_catalogs/buzzard_base/config_file.para  -t G   --GAL_LIB_IN buzzard_SEDs   --GAL_LIB_OUT buzzard_LSST   --FILTER_FILE photozdc1   --MAGTYPE AB   --MOD_EXTINC 0    --EXTINC_LAW calzetti.dat   --EM_DISPERSION 0   --LIB_ASCII YES   --Z_STEP 0.01,0,2

###buzzard in DES filters (for DESstars, star-gal)
mag_gal  -c lephare/training_stats/simulation_catalogs/buzzard_base/config_file.para  -t G   --GAL_LIB_IN buzzard_SEDs   --GAL_LIB_OUT buzzard_DES   --FILTER_FILE photozDES   --MAGTYPE AB   --MOD_EXTINC 0    --EXTINC_LAW calzetti.dat   --EM_DISPERSION 0   --LIB_ASCII YES   --Z_STEP 0.01,0,2

###PICKLES in DES filters (for DESstars, only star)
mag_gal  -c lephare/training_stats/simulation_catalogs/DES/DES_STARCAT/config_file.para   -t S   --STAR_LIB_IN PICKLES_SEDs   --STAR_LIB_OUT PICKLES_DES   --FILTER_FILE photozDES   --MAGTYPE AB   --MOD_EXTINC 0    --EXTINC_LAW calzetti.dat   --EM_DISPERSION 0   --LIB_ASCII YES   --Z_STEP 0,0,0

###PICKLES in LSST filters (for DC1, star-gal)
mag_gal  -c lephare/training_stats/simulation_catalogs/DES/DES_STARCAT/config_file.para   -t S   --STAR_LIB_IN PICKLES_SEDs   --STAR_LIB_OUT PICKLES_LSST   --FILTER_FILE photozdc1   --MAGTYPE AB   --MOD_EXTINC 0    --EXTINC_LAW calzetti.dat   --EM_DISPERSION 0   --LIB_ASCII YES   --Z_STEP 0,0,0



##### zphota (run) #####

###DC1 - Compute redshifts
zphota -c lephare/training_stats/simulation_catalogs/buzzard_base/config_file.para  --CAT_IN lephare/training_stats/simulation_catalogs/buzzard_base/Shuffle_Buzzard_training_file.dat   --INP_TYPE M   --CAT_TYPE LONG   --CAT_MAG AB   --PARA_OUT lephare/training_stats/simulation_catalogs/buzzard_base/output_DC1.para  --CAT_OUT lephare/training_stats/simulation_catalogs/buzzard_base/zphota/DC1_buzzard_LSST.out  --ZPHOTLIB buzzard_LSST   --AUTO_ADAPT NO  --CAT_FMT MMEE   --ERR_SCALE 0  --CAT_LINES 0,100000  --PDZ_OUT lephare/training_stats/simulation_catalogs/buzzard_base/zphota/PDZs/DC1_buzzard_LSST  --PDZ_TYPE MIN_ZG


###DESstars - check star SED fitting
zphota -c lephare/training_stats/simulation_catalogs/DES/DES_STARCAT/config_file.para  --CAT_IN lephare/training_stats/simulation_catalogs/DES/DES_STARCAT/Y6A1_FGCM_V3_3_1_PSF_ALL_STARS_shuffled.dat   --INP_TYPE M   --CAT_TYPE SHORT   --CAT_MAG AB   --PARA_OUT lephare/training_stats/simulation_catalogs/DES/DES_STARCAT/WORK_COMPLETE2/zphota/output_STAR.para  --CAT_OUT lephare/training_stats/simulation_catalogs/DES/DES_STARCAT/WORK_COMPLETE2/zphota/DESstar_PICKLES_DES.out  --ZPHOTLIB PICKLES_DES   --AUTO_ADAPT NO  --CAT_FMT MMEE   --ERR_SCALE 0  --CAT_LINES 0,100000  --STAR_PDF_OUT lephare/training_stats/simulation_catalogs/DES/DES_STARCAT/WORK_COMPLETE2/zphota/star_pdfs/DESstar_PICKLES_DES
